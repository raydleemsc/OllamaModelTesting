\section{Introduction to Topology}

Topology is a branch of mathematics that studies properties of spaces that are preserved under continuous deformations. This cheat sheet covers basic definitions, theorems, and corollaries in topology.

\section{Definitions}

\begin{definition}[Topological Space]
A \emph{topological space} $(X, \tau)$ consists of a set $X$ and a collection $\tau$ of subsets of $X$, called \emph{open sets}, such that:
\begin{enumerate}
    \item The empty set $\emptyset$ and the entire set $X$ are in $\tau$.
    \item Any union of open sets is an open set.
    \item Any finite intersection of open sets is an open set.
\end{enumerate}
\end{definition}

\begin{definition}[Closed Set]
A subset $C$ of a topological space $(X, \tau)$ is called \emph{closed} if its complement $X \setminus C$ is open.
\end{definition}

\begin{definition}[Interior and Closure]
Let $(X, \tau)$ be a topological space and $A \subseteq X$.
\begin{enumerate}
    \item The \emph{interior} of $A$, denoted by $\text{int}(A)$, is the union of all open sets contained in $A$.
    \item The \emph{closure} of $A$, denoted by $\overline{A}$, is the intersection of all closed sets containing $A$.
\end{enumerate}
\end{definition}

\begin{definition}[Base and Subbase]
Let $(X, \tau)$ be a topological space.
\begin{enumerate}
    \item A \emph{base} for $\tau$ is a collection $\mathcal{B}$ of open sets such that every open set in $\tau$ can be written as a union of elements from $\mathcal{B}$.
    \item A \emph{subbase} for $\tau$ is a collection $\mathcal{S}$ of subsets of $X$ whose finite intersections form a base for $\tau$.
\end{enumerate}
\end{definition}

\begin{definition}[Continuous Function]
Let $(X, \tau_X)$ and $(Y, \tau_Y)$ be topological spaces. A function $f: X \to Y$ is called \emph{continuous} if for every open set $V \in \tau_Y$, the preimage $f^{-1}(V) \in \tau_X$.
\end{definition}

\begin{definition}[Homeomorphism]
A bijective continuous function $f: X \to Y$ between topological spaces is called a \emph{homeomorphism} if its inverse $f^{-1}: Y \to X$ is also continuous. Two spaces are \emph{homeomorphic} if there exists a homeomorphism between them.
\end{definition}

\section{Theorems and Corollaries}

\begin{theorem}[Characterization of Open Sets]
Let $(X, \tau)$ be a topological space. A subset $U \subseteq X$ is open if and only if for every $x \in U$, there exists an open set $V \in \tau$ such that $x \in V \subseteq U$.
\end{theorem}

\begin{corollary}[Interior of a Set]
For any subset $A$ of a topological space $(X, \tau)$, the interior $\text{int}(A)$ is the largest open set contained in $A$.
\end{corollary}

\begin{theorem}[Characterization of Closed Sets]
Let $(X, \tau)$ be a topological space. A subset $C \subseteq X$ is closed if and only if its complement $X \setminus C$ is open.
\end{theorem}

\begin{corollary}[Closure of a Set]
For any subset $A$ of a topological space $(X, \tau)$, the closure $\overline{A}$ is the smallest closed set containing $A$.
\end{corollary}

\begin{theorem}[Continuity and Open Sets]
Let $(X, \tau_X)$ and $(Y, \tau_Y)$ be topological spaces. A function $f: X \to Y$ is continuous if and only if for every open set $V \in \tau_Y$, the preimage $f^{-1}(V) \in \tau_X$.
\end{theorem}

\begin{corollary}[Continuity and Closed Sets]
Let $(X, \tau_X)$ and $(Y, \tau_Y)$ be topological spaces. A function $f: X \to Y$ is continuous if and only if for every closed set $C \in \tau_Y$, the preimage $f^{-1}(C)$ is closed in $\tau_X$.
\end{corollary}

\begin{theorem}[Homeomorphism Theorem]
Let $(X, \tau_X)$ and $(Y, \tau_Y)$ be topological spaces. If there exists a homeomorphism $f: X \to Y$, then the topological properties of $X$ are preserved in $Y$.
\end{theorem}

\section{Examples}

\begin{itemize}
    \item The standard topology on $\mathbb{R}$ is generated by the base consisting of open intervals $(a, b)$.
    \item The discrete topology on any set $X$ is the topology where every subset of $X$ is open.
    \item The trivial (or indiscrete) topology on a set $X$ consists only of $\emptyset$ and $X$ itself.
\end{itemize}
